local FPSSettings = {
    RemoveTextures = false,
    RemoveParticles = false,
    RemoveDecals = false,
    OptimizeLighting = false,
    ReduceShadows = false,
    DisableBloom = false,
    DisableBlur = false,
    DisableSunRays = false,
    OptimizeTerrain = false,
    ReduceRenderDistance = false
}

local originalSettings = {}

-- Function to save original settings
local function saveOriginalSettings()
    originalSettings.Brightness = Lighting.Brightness
    originalSettings.GlobalShadows = Lighting.GlobalShadows
    originalSettings.Technology = Lighting.Technology
end

-- Ultra FPS Optimizer
local function applyFPSBoost()
    pcall(function()
        -- Optimize Graphics
        settings().Rendering.QualityLevel = 1
        
        -- Remove Textures
        if FPSSettings.RemoveTextures then
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("Texture") or v:IsA("Decal") then
                    v.Transparency = 1
                end
                if v:IsA("MeshPart") or v:IsA("Part") then
                    v.Material = Enum.Material.SmoothPlastic
                    v.Reflectance = 0
                end
            end
        end
        
        -- Remove Particles
        if FPSSettings.RemoveParticles then
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Smoke") or v:IsA("Fire") or v:IsA("Sparkles") then
                    v.Enabled = false
                end
            end
        end
        
        -- Optimize Lighting
        if FPSSettings.OptimizeLighting then
            Lighting.GlobalShadows = false
            Lighting.Brightness = 2
            Lighting.Technology = Enum.Technology.Compatibility
        end
        
        -- Remove Post Effects
        if FPSSettings.DisableBloom then
            for _, effect in pairs(Lighting:GetChildren()) do
                if effect:IsA("BloomEffect") then
                    effect.Enabled = false
                end
            end
        end
        
        if FPSSettings.DisableBlur then
            for _, effect in pairs(Lighting:GetChildren()) do
                if effect:IsA("BlurEffect") then
                    effect.Enabled = false
                end
            end
        end
        
        if FPSSettings.DisableSunRays then
            for _, effect in pairs(Lighting:GetChildren()) do
                if effect:IsA("SunRaysEffect") then
                    effect.Enabled = false
                end
            end
        end
        
        -- Optimize Terrain
        if FPSSettings.OptimizeTerrain then
            workspace.Terrain.Decoration = false
        end
    end)
end

local function restoreGraphics()
    pcall(function()
        -- Restore Textures
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("Texture") or v:IsA("Decal") then
                v.Transparency = 0
            end
        end
        
        -- Restore Particles
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("ParticleEmitter") or v:IsA("Trail") then
                v.Enabled = true
            end
        end
        
        -- Restore Lighting
        if originalSettings.Brightness then
            Lighting.GlobalShadows = originalSettings.GlobalShadows
            Lighting.Brightness = originalSettings.Brightness
            Lighting.Technology = originalSettings.Technology
        end
        
        -- Restore Effects
        for _, effect in pairs(Lighting:GetChildren()) do
            if effect:IsA("BloomEffect") or effect:IsA("BlurEffect") or effect:IsA("SunRaysEffect") then
                effect.Enabled = true
            end
        end
        
        -- Restore Terrain
        workspace.Terrain.Decoration = true
    end)
end

local function createESP(item, color)
    if item:FindFirstChild("ESP") then return end

    local h = Instance.new("Highlight")
    h.Name = "ESP"
    h.FillColor = color
    h.OutlineColor = color
    h.FillTransparency = 0.5
    h.Parent = item

    local bill = Instance.new("BillboardGui")
    bill.Name = "ESPLabel"
    bill.Size = UDim2.new(0, 140, 0, 40)
    bill.AlwaysOnTop = true
    bill.StudsOffset = Vector3.new(0, 2, 0)
    bill.Parent = item

    local text = Instance.new("TextLabel")
    text.Size = UDim2.new(1, 0, 1, 0)
    text.BackgroundTransparency = 1
    text.TextScaled = true
    text.TextColor3 = color
    text.Font = Enum.Font.SourceSansBold
    text.TextStrokeTransparency = 0
    text.Text = item.Name
    text.Parent = bill
end

local function scanExact(name, color)
    if not itemsFolder then return end
    for _, item in pairs(itemsFolder:GetChildren()) do
        if string.lower(item.Name) == string.lower(name) then
            createESP(item, color)
        end
    end
end

local function clearESP()
    if not itemsFolder then return end
    for _, item in pairs(itemsFolder:GetChildren()) do
        if item:FindFirstChild("ESP") then item.ESP:Destroy() end
        if item:FindFirstChild("ESPLabel") then item.ESPLabel:Destroy() end
    end
end

-- SPEED LOCK
local speedValue = 50
local speedConn

local function applySpeed(char)
    local hum = char:WaitForChild("Humanoid")
    hum.WalkSpeed = speedValue

    if speedConn then speedConn:Disconnect() end
    speedConn = hum:GetPropertyChangedSignal("WalkSpeed"):Connect(function()
        if hum.WalkSpeed ~= speedValue then
            hum.WalkSpeed = speedValue
        end
    end)
end

if player.Character then applySpeed(player.Character) end
player.CharacterAdded:Connect(applySpeed)

PlayerTab:CreateInput({
    Name = "WalkSpeed",
    PlaceholderText = "Enter speed",
    Callback = function(t)
        local n = tonumber(t)
        if n then
            speedValue = n
            local c = player.Character
            if c and c:FindFirstChild("Humanoid") then
                c.Humanoid.WalkSpeed = speedValue
            end
        end
    end
})

local killAura = false
local auraRange = 25
local selectedWeaponName = nil
local useSelectedWeapon = false

local function getHeldTool()
    local char = player.Character
    if not char then return end
    for _, v in pairs(char:GetChildren()) do
        if v:IsA("Tool") then return v end
    end
end

local function getWeapon(char)
    if useSelectedWeapon and selectedWeaponName then
        for _, tool in pairs(char:GetChildren()) do
            if tool:IsA("Tool") and tool.Name == selectedWeaponName then
                return tool
            end
        end
    end
    for _, tool in pairs(char:GetChildren()) do
        if tool:IsA("Tool") then return tool end
    end
end

local ZombieList = {
    spitter = true, bloater = true, hazmat = true, muscle = true,
    runner = true, crawler = true, zombie = true, riot = true
}

local function isZombie(m)
    return ZombieList[string.lower(m.Name)] ~= nil
end

local function getTargets()
    local t = {}
    local char = player.Character
    if not char then return t end

    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return t end

    local charactersFolder = workspace:FindFirstChild("Characters")
    if not charactersFolder then return t end

    for _, v in pairs(charactersFolder:GetChildren()) do
        if v:IsA("Model") and isZombie(v) then
            local hum = v:FindFirstChildOfClass("Humanoid")
            local part = v:FindFirstChild("HumanoidRootPart")

            if hum and part and hum.Health > 0 then
                if (hrp.Position - part.Position).Magnitude <= auraRange then
                    table.insert(t, v)
                end
            end
        end
    end
    return t
end

task.spawn(function()
    while task.wait(0.15) do
        if killAura then
            local char = player.Character
            if not char then continue end

            local tool = getWeapon(char)
            local auto = char:FindFirstChild("AutoTargetClient")

            if tool and auto then
                local targets = getTargets()
                if #targets > 0 then
                    pcall(function()
                        auto.UpdateNearbyTargets:FireServer(targets)
                        if tool:FindFirstChild("HitTargets") then
                            tool.HitTargets:FireServer(targets)
                        end
                        if tool:FindFirstChild("Swing") then
                            tool.Swing:FireServer()
                        end
                    end)
                end
            end
        end
    end
end)

ExploitsTab:CreateSection("Auto Pickup")

local autoPickupActive = false
local autoPickupRadius = 15
local autoPickupAllItems = false
local autoPickupSelected = {}
local autoPickupConnection = nil

local droppedItemsFolder = workspace:FindFirstChild("DroppedItems")

local function getItemMainPart(model)
    return model:FindFirstChildWhichIsA("BasePart")
end

local function startAutoPickup()
    if autoPickupConnection then autoPickupConnection:Disconnect() end

    autoPickupConnection = RunService.Heartbeat:Connect(function()
        if not autoPickupActive then return end

        local char = player.Character
        if not char then return end

        local root = char:FindFirstChild("HumanoidRootPart")
        if not root or not droppedItemsFolder then return end

        local pos = root.Position

        for _, item in ipairs(droppedItemsFolder:GetChildren()) do
            if not (autoPickupAllItems or autoPickupSelected[item.Name]) then continue end

            local part = item.PrimaryPart or getItemMainPart(item)

            if part and (part.Position - pos).Magnitude <= autoPickupRadius then
                pcall(function()
                    if adjustBackpackRemote then
                        adjustBackpackRemote:FireServer(item)
                    elseif pickUpItemRemote then
                        pickUpItemRemote:FireServer(item)
                    end
                end)
                task.wait(0.05)
            end
        end
    end)
end

ExploitsTab:CreateToggle({
    Name = "Auto Pickup",
    Callback = function(v)
        autoPickupActive = v
        if v then 
            startAutoPickup()
        elseif autoPickupConnection then 
            autoPickupConnection:Disconnect() 
        end
    end
})

ExploitsTab:CreateToggle({
    Name = "Pickup All Items",
    Callback = function(v) autoPickupAllItems = v end
})

ExploitsTab:CreateSlider({
    Name = "Pickup Radius",
    Range = {5, 100},
    Increment = 1,
    CurrentValue = 15,
    Callback = function(v) autoPickupRadius = v end
})

local noclip = false
local noclipConn

local function startNoclip()
    if noclipConn then noclipConn:Disconnect() end
    noclipConn = RunService.Stepped:Connect(function()
        if noclip and player.Character then
            for _, part in pairs(player.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end
        end
    end)
end

player.CharacterAdded:Connect(function()
    if noclip then
        task.wait(1)
        startNoclip()
    end
end)

PlayerTab:CreateToggle({
    Name = "Noclip",
    CurrentValue = false,
    Callback = function(v)
        noclip = v
        if v then
            startNoclip()
        else
            if noclipConn then noclipConn:Disconnect() end
            if player.Character then
                for _, part in pairs(player.Character:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.CanCollide = true
                    end
                end
            end
        end
    end
})
